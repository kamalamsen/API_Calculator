import pandas as pd
import streamlit as st
from io import BytesIO

# -------------------- COMMON UTILITIES --------------------
def normalize_headers(df):
    df.columns = df.columns.str.strip().str.lower()
    return df

# -------------------- API LOGIC (COMMON) --------------------
def calculate_api_from_percentage(percentages):
    categories = [
        (95, 100, 10),
        (90, 94.99, 8),
        (80, 89.99, 6),
        (70, 79.99, 4),
        (60, 69.99, 2),
        (50, 59.99, 0),
        (33, 49.99, -1),
        (0, 32.99, -3),
    ]
    total_weighted_score = 0
    for pct in percentages:
        for low, high, weight in categories:
            if low <= pct <= high:
                total_weighted_score += weight
                break
    return (total_weighted_score / len(percentages)) * 100

# -------------------- SINGLE SUBJECT API --------------------
def calculate_single_subject_api(file):
    df = normalize_headers(pd.read_excel(file))

    if not {'name', 'marks'}.issubset(df.columns):
        st.error("Excel must contain columns: Name, Marks")
        return

    if df['marks'].max() > 100 or df['marks'].min() < 0:
        st.error("Marks must be between 0 and 100")
        return

    df['percentage'] = df['marks']
    # Ranking only for single subject
    df['rank'] = df['percentage'].rank(ascending=False, method='dense').astype(int)

    api_score = calculate_api_from_percentage(df['percentage'])

    st.subheader("Single Subject API Result")
    # ---- Editable Remarks Section ----
st.subheader("Editable Teacher Remarks (Optional)")
remarks_df = df[['name', 'teacher remark']].copy()
edited_remarks = st.data_editor(remarks_df, num_rows="dynamic")

# Update remarks back to main dataframe
for i, row in edited_remarks.iterrows():
    df.loc[df['name'] == row['name'], 'teacher remark'] = row['teacher remark']

st.subheader("Analysis Result")
st.dataframe(df)
    st.write(f"Class API Score: {api_score:.2f}")

    # ---- DIVISION-WISE COUNT TABLE (Percentage Ranges) ----
    st.subheader("Division-wise Distribution (Percentage Ranges)")

    def division_bucket(pct):
        if pct >= 95:
            return '>95'
        elif pct >= 90:
            return '90-94.99'
        elif pct >= 80:
            return '80-89.99'
        elif pct >= 70:
            return '70-79.99'
        elif pct >= 60:
            return '60-69.99'
        elif pct >= 50:
            return '50-59.99'
        elif pct >= 33:
            return '33-49.99'
        else:
            return '<33'

    df['division'] = df['percentage'].apply(division_bucket)
    division_count_df = df['division'].value_counts().reset_index()
    division_count_df.columns = ['Division', 'Count']
    st.dataframe(division_count_df)

    # ---- PERFORMANCE CATEGORY TABLE ----
    st.subheader("Performance Category Count")

    def performance_tag(pct):
        if pct >= 95:
            return 'High Achiever'
        elif pct >= 75:
            return 'Average'
        elif pct >= 50:
            return 'Needs Improvement'
        elif pct >= 33:
            return 'Remedial'
        else:
            return 'Critical'

    df['performance category'] = df['percentage'].apply(performance_tag)

    # ---- AUTOMATIC REMEDIAL SUGGESTION ----
    def remedial_suggestion(category):
        if category == 'High Achiever':
            return 'Provide enrichment tasks and higher-order questions'
        elif category == 'Average':
            return 'Regular practice with exam-oriented questions'
        elif category == 'Needs Improvement':
            return 'Focused revision and topic-wise worksheets'
        elif category == 'Remedial':
            return 'Small group remedial sessions and basics reinforcement'
        else:
            return 'Immediate intervention and personalised mentoring'

    df['suggested action'] = df['performance category'].apply(remedial_suggestion)

    # ---- AUTO-GENERATED TEACHER REMARKS ----
    def teacher_remark(row):
        if row['performance category'] == 'High Achiever':
            return 'Excellent performance. Keep up the consistent effort.'
        elif row['performance category'] == 'Average':
            return 'Good effort. With focused practice, further improvement is possible.'
        elif row['performance category'] == 'Needs Improvement':
            return 'Needs more practice and regular revision to strengthen concepts.'
        elif row['performance category'] == 'Remedial':
            return 'Requires remedial support and close academic monitoring.'
        else:
            return 'Immediate attention required. Individual guidance and parental support recommended.'

    df['teacher remark'] = df.apply(teacher_remark, axis=1)

    # ---- SUBJECT-WISE WEAKNESS REMARK (AUTO) ----
    def weakness_remark(pct):
        if pct >= 75:
            return 'Concepts are clear. Continue with advanced practice.'
        elif pct >= 50:
            return 'Some topics need reinforcement through practice.'
        elif pct >= 33:
            return 'Basic concepts need strengthening.'
        else:
            return 'Major conceptual gaps observed.'

    df['subject-wise remark'] = df['percentage'].apply(weakness_remark)

    # ---- BOARD-FOCUSED REMARK (CLASS 9 & 10) ----
    def board_remark(pct):
        if pct >= 90:
            return 'Well prepared for board-level questions.'
        elif pct >= 75:
            return 'On track for boards with regular revision.'
        elif pct >= 50:
            return 'Needs structured board-oriented practice.'
        else:
            return 'Requires intensive board exam preparation support.'

    df['board exam remark'] = df['percentage'].apply(board_remark)
    category_count_df = df['performance category'].value_counts().reset_index()
    category_count_df.columns = ['Category', 'Count']
    st.dataframe(category_count_df)

    # ---- STUDENT LIST BY CATEGORY ----
    st.subheader("Students Grouped by Performance Category")
    for category in category_count_df['Category']:
        st.markdown(f"**{category}**")
        st.dataframe(df[df['performance category'] == category][['name', 'percentage', 'rank']])

    output = BytesIO()
    with pd.ExcelWriter(output, engine='xlsxwriter') as writer:
        # Main processed data
        df.to_excel(writer, sheet_name='Student Analysis', index=False)

        # API summary
        pd.DataFrame({'API Score': [api_score]}).to_excel(writer, sheet_name='Summary', index=False)

        # Division-wise summary
        division_count_df.to_excel(writer, sheet_name='Division Summary', index=False)

        # Category-wise summary
        category_count_df.to_excel(writer, sheet_name='Category Summary', index=False)

        # Student lists per category
        for cat in category_count_df['Category']:
            df[df['performance category'] == cat].to_excel(
                writer,
                sheet_name=f"{cat[:25]}",
                index=False
            )

    st.download_button(
        "Download Final Excel",
        output.getvalue(),
        "API_Single_Subject.xlsx",
        "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet"
    )

# -------------------- FIVE SUBJECT API --------------------
def calculate_five_subject_api(file):
    df = normalize_headers(pd.read_excel(file))
    subject_cols = ['subject1', 'subject2', 'subject3', 'subject4', 'subject5']

    if not all(col in df.columns for col in ['name'] + subject_cols):
        st.error("Excel must contain Name and Subject1 to Subject5")
        return

    if df[subject_cols].max().max() > 100 or df[subject_cols].min().min() < 0:
        st.error("Marks must be between 0 and 100")
        return

    df['total'] = df[subject_cols].sum(axis=1)
    df['percentage'] = (df['total'] / 500) * 100

    # Performance category for five-subject API
    def performance_tag(pct):
        if pct >= 95:
            return 'High Achiever'
        elif pct >= 75:
            return 'Average'
        elif pct >= 50:
            return 'Needs Improvement'
        elif pct >= 33:
            return 'Remedial'
        else:
            return 'Critical'

    df['performance category'] = df['percentage'].apply(performance_tag)

    # Ranking for five-subject API
    df['rank'] = df['percentage'].rank(ascending=False, method='dense').astype(int)

    api_score = calculate_api_from_percentage(df['percentage'])

    st.subheader("Five Subject API Result")
    st.dataframe(df)
    st.write(f"Class API Score: {api_score:.2f}")

    # ---- DIVISION-WISE COUNT TABLE (Percentage Ranges) ----
    st.subheader("Division-wise Distribution (Percentage Ranges)")

    def division_bucket(pct):
        if pct >= 95:
            return '>95'
        elif pct >= 90:
            return '90-94.99'
        elif pct >= 80:
            return '80-89.99'
        elif pct >= 70:
            return '70-79.99'
        elif pct >= 60:
            return '60-69.99'
        elif pct >= 50:
            return '50-59.99'
        elif pct >= 33:
            return '33-49.99'
        else:
            return '<33'

    df['division'] = df['percentage'].apply(division_bucket)
    division_count_df = df['division'].value_counts().reset_index()
    division_count_df.columns = ['Division', 'Count']
    st.dataframe(division_count_df)

    # ---- CATEGORY COUNT TABLE (High Achiever, etc.) ----
    st.subheader("Performance Category Count")
    category_count_df = df['performance category'].value_counts().reset_index()
    category_count_df.columns = ['Category', 'Count']
    st.dataframe(category_count_df)

    # ---- Student-wise Tables by Category ----
    st.subheader("Students Grouped by Performance Category")
    for category in category_count_df['Category']:
        st.markdown(f"**{category}**")
        st.dataframe(df[df['performance category'] == category][['name', 'percentage', 'rank']])

    output = BytesIO()
    with pd.ExcelWriter(output, engine='xlsxwriter') as writer:
        df.to_excel(writer, sheet_name='Processed Data', index=False)
        pd.DataFrame({'API Score': [api_score]}).to_excel(writer, sheet_name='Summary', index=False)

    st.download_button(
        "Download Final Excel",
        output.getvalue(),
        "API_Five_Subjects.xlsx",
        "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet"
    )
